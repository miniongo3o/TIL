import java.util.Scanner;
public class Main{
    public static void main(String[] args){
		Scanner sc=new Scanner(System.in);
        int num=sc.nextInt();
        int target=num;
        int cnt=1;
        
        while(true){
            int ten=num%10;
            int one=(num/10+num%10)%10;
            num=ten*10+one;
            
            
            
            if(num==target){
                break;
            }
            cnt++;
        }
        System.out.println(cnt);
        
        
    }
}
